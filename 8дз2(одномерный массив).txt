#include <iostream>
#include <string>
using namespace std;

int main()
{
    setlocale(LC_ALL, "rus");
    srand(time(NULL));
    const int a = 20;
    int arr[a];
   
    for(int i=0;i<a;i++)
    {
       arr[i] =rand() % 10;
        cout << arr[i] << " ";
    }
    cout << "\n\n";
    for(int i=0;i<a;i++)
    {
        if (arr[i] == 0) 
        {
            arr[i] = -1; 
        }
        cout << arr[i] << " ";
    }
    
    return 0;
}
